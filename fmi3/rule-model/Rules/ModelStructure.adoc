validDependencyKinds: Location * [seq of (nat)] * [seq of (DependenciesKind)] +> bool

	dependenciesKind <> nil =>
		-- @OnFail("validDependencyKinds: dependencies do not match dependenciesKind at %s", loc2str(location))
